124  template <class T>
125  T* cast(const std::string& requestor = "Unknown")
126  {
127  const std::type_info& ti = typeid(T) ;
128  Component * target = this ;
129  //To avoid possibility of infinite loops we cap this protocol at two attempts
130  for (int iTry=0; iTry<2 && target; ++iTry) {
131  //First see if the Component is a T
132  T* pT = dynamic_cast<T*>(target) ;
133  if (pT) //Okay I'm a T!
134  return pT ;
135  //Otherwise maybe I own a T. My implementor is then responsible for providing
136  target = target->acquire(ti, reinterpret_cast<void **>(&pT)) ;
137  if (pT) //Yay, i'm bound.
138  return pT ;
139  }
140  throw RUNTIME_ERROR("Can't cast Component [%s] to type [%s] as required by [%s]", name().c_str(), ti.name(), requestor.c_str()) ;
141  }
