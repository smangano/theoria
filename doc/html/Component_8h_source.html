244  template <class T>
245  T* cast(const std::string& requestor = "Unknown")
246  {
247  const std::type_info& ti = typeid(T) ;
248  Component * target = this ;
249  //To avoid possibility of infinite loops we cap this protocol at two attempts
250  for (int iTry=0; iTry<2 && target; ++iTry) {
251  //First see if the Component is a T
252  T* pT = dynamic_cast<T*>(target) ;
253  if (pT) //Okay I'm a T!
254  return pT ;
255  //Otherwise maybe I own a T. My implementor is then responsible for providing
256  target = target->acquire(ti, reinterpret_cast<void **>(&pT)) ;
257  if (pT) //Yay, i'm bound.
258  return pT ;
259  }
260  throw RUNTIME_ERROR("Can't cast Component [%s] to type [%s] as required by [%s]", name().c_str(), ti.name(), requestor.c_str()) ;
261  }
